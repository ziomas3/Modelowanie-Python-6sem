import random
import numpy as np

def init(p,sc):
    #losowa obsada sieci
    for k in range (0,L):
        for i in range(0,L):
            if random.random()<p:
                sc[k,i]=-1
            else:
                sc[k,i]=0
    czerw=[]
    for i in range(0,L):
        if sc[0,i]==-1:
            czerw.append([0,i]) #dodajemy do czerw
            sc[0,i]=1 
    return czerw
            
def krok_propag(sc,czerw):
    tymcz=[]
    for wz in czerw:
        if wz[0]>0 and sc[wz[0]-1,wz[1]]==-1: #(-1,0)
            tymcz.append([wz[0]-1,wz[1]])
            sc[wz[0]-1,wz[1]]=1
        if wz[0]<Lm1 and sc[wz[0]+1,wz[1]]==-1: #(1,0)
            tymcz.append([wz[0]+1,wz[1]])
            sc[wz[0]+1,wz[1]]=1
        if wz[1]>0 and sc[wz[0],wz[1]-1]==-1: #(0,-1)
            tymcz.append([wz[0],wz[1]-1])
            sc[wz[0],wz[1]-1]=1
        if wz[1]<Lm1 and sc[wz[0],wz[1]+1]==-1: #(0,1)
            tymcz.append([wz[0],wz[1]+1])
            sc[wz[0],wz[1]+1]=1 
    return tymcz

def init_siec(p,sc):
    #losowa obsada sieci
    for k in range (0,L):
        for i in range(0,L):
            if random.random()<p:
                sc[k,i]=-1
            else:
                sc[k,i]=0
    zainfek=[]
    for i in range(0,L):
        if sc[0,i]==-1:
            zainfek.append([0,i]) #dodajemy do zainfek
            sc[0,i]=1 
    return zainfek
            
def krok_propag(sc,czerw):
    tymcz=[]
    jest=0
    for wz in czerw:
        if wz[0]>0 and sc[wz[0]-1,wz[1]]==-1: #(-1,0)
            tymcz.append([wz[0]-1,wz[1]])
            sc[wz[0]-1,wz[1]]=1
        if wz[0]<Lm1 and sc[wz[0]+1,wz[1]]==-1: #(1,0)
            tymcz.append([wz[0]+1,wz[1]])
            sc[wz[0]+1,wz[1]]=1          
            if (wz[0]+1)==Lm1:               #czy dotar do brzegu
                jest=1
                break
        if wz[1]>0 and sc[wz[0],wz[1]-1]==-1: #(0,-1)
            tymcz.append([wz[0],wz[1]-1])
            sc[wz[0],wz[1]-1]=1
        if wz[1]<Lm1 and sc[wz[0],wz[1]+1]==-1: #(0,1)
            tymcz.append([wz[0],wz[1]+1])
            sc[wz[0],wz[1]+1]=1 
    return jest,tymcz
