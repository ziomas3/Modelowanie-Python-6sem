Dyfuzyjny wzrost zlepków¶

Symulacje dyfuzyjnego wzrostu zlepka moemy przeprowadzi na sieci kwadratowej w nastpujcy sposb

\begin{itemize} \item W rodku sieci umieszczamy zarodek zlepka. \item Cztery ssiednie wzy wok zarodka tworz pocztkowy obwd zlepka. \item Do zlepka bd doczane czstki, ktre poruszajc si losowo po wzach sieci dotr do obwodu zlepka. \item Miejscem startu kadej z czstek moe by losowe pooenie na okrgu o promieniu R, ktrego rodek ley w rodku sieci. \item Wzy brzegowe sieci mona potraktowa jako wzy pochaniajce, co powoduje, e losowy spacer kadej z czstek moe zakoczy si albo na brzegu sieci albo doczeniem do zlepka. \item Symulacj koczymy gdy zlepek osignie zadan z gry mas (liczb czstek) lub gdy jego rozmiar liniowy bdzie bliski 2R. \end{itemize}

import random
import numpy as np
import matplotlib.pyplot as plt

v=np.array([[1, 0],[0, 1],[-1, 0],[0,-1]]) # 4 wektory do najbliszych ssiadw

Lp=300  # wsprzdna rodka sieci 
L=2*Lp  # rozmiar liniowy sieci
siec=np.zeros((L,L),int)  # siec o  L*L pocatkowo wyzerowana
#  Skadowe sieci mog przyjmowa wartoci -1, 0,1 i 2
# Wartoci te reprezentuj
#  -1 wzy obwodu
#   0 wzy dyfuzyjne
#   1 wzy zlepka
#   2 wzy pochlaniajce

def init():
    siec[Lp,Lp]=1  #zarodek
    mod_obwodu(Lp,Lp)
    #wezly pochlaniajace
    for i in range(L):
        siec[i,0]=2
        siec[i,L-1]=2
        siec[0,i]=2
        siec[L-1,i]=2
    return np.array([Lp,Lp],int)


def start_cz(R):
    kat=2*np.pi*random.random()
    # losowe pooenie pocztkowe na okrgu o promieniu R
    return np.array([Lp+np.rint(R*np.cos(kat)),Lp+np.rint(R*np.sin(kat))],int)
       
def mod_obwodu(x,y):
    for i in range(4):
        if siec[x+v[i,0],y+v[i,1]]==0:    
            siec[x+v[i,0],y+v[i,1]]=-1

def spacer_los(R): 
    #pooenie pocztkowe na okrgu o promieniu R
    r=start_cz(R)
    while siec[r[0],r[1]]==0:
        r=r+v[random.randint(0,3)] #losowy  krok
    return siec[r[0],r[1]], r 

def wzrost(R,mmax,xy_z):
    licz=1
    rb=init()
    xy_z.append(rb)
    while licz<mmax:        
        status,rb=spacer_los(R)
        if status==-1:
            siec[rb[0],rb[1]]=1
            mod_obwodu(rb[0],rb[1])
            xy_z.append(rb)
            licz+=1
        
    

def rys_zlep(kn,xy):
    fig, ax=plt.subplots()
    xt, yt = zip(*xy)
    xf=xt[:kn]
    yf=yt[:kn]
    mn=min(min(xf),min(yf))
    mx=max(max(xf),max(yf))
    dm=np.rint((mx-mn)/4)
    if dm <1:
        dm=1
    ax.set_xticks(np.arange(mn-dm,mx+dm,dm))
    ax.set_yticks(np.arange(mn-dm,mx+dm,dm))
    zlep=ax.scatter(xf,yf,s=6,c='r')
    ax.set_ylabel('y')
    ax.set_xlabel('x')
    tytul = "Zlepek o masie =" +str(kn)
    ax.set_title(tytul)
    plt.axis('equal')
    plt.show()

Wzrost zlepka na obwodzie koła.¶

Zarodki rozmieszczone s na okrgu o promieniu R. Kolejne czstki startuj ze rodka koa, poruszajc si losowo po wzach sieci. Gdy czstka dotrze do obwodu zlepka zostaje do niego przyczona.

#   1 wzy zlepka


def initC(Rc):
    Rc2=Rc**2
    for i in range(LC):
        for k in range(LC):
            if ((i-LCp)**2+(k-LCp)**2)>=Rc2:
                circle[i,k]=1  #zarodek
    for i in range(1,LC-1):
        for k in range(1,LC-1):
            if n_sas(i,k)>0:
                circle[i,k]=-1
    
def n_sas(x,y):
    licz=0
    if circle[x,y]==0:
        for i in range(4):
            if circle[x+v[i,0],y+v[i,1]]==1:
                licz+=1
    return licz
        

def start_C():
    return np.array([LCp,LCp],int)
       
def mod_obwC(x,y):

Model Edena¶

Prosty model zaproponowany przez Edena do symulacji wzrostu kolonii komrkowych.

W tym modelu zaczynamy od umieszczenia zarodka w rodku sieci. Niezajte ssiednie wzy tworz obwd zlepka. W najprostszej wersji modelu nowa czstka jest doczana do losowo wybranego wza z obwodu zlepka. Doczenie nowej czstki modyfikuje obwd. Taki proces wzrostu jest powtarzany tak dugo a zlepek osignie brzeg sieci.

def rys_zlep(kn):
#4 wektory do najbliszych ssiadw
    v=np.array([[1, 0],[0, 1],[-1, 0],[0,-1]])
    Lp=500 
    L=2*Lp
    siec=np.zeros((L,L),int)

    zlep_x=[]
    zlep_y=[]
#zarodek
    r=np.array([0,0])
    lista=[[Lp,Lp]]
    for k in range (kn):
        r=random.choice(lista)
        if (r[0]==0)or(r[0]==L-1):
            break
        if (r[1]==0)or(r[1]==L-1):
            break    
        siec[r[0],r[1]]=1
        zlep_x.append(r[0])
        zlep_y.append(r[1])
        for i in range(4):
            rs=r+v[i]
            if(siec[rs[0],rs[1]]==0):
                lista.append([rs[0],rs[1]])
        lista.remove([r[0],r[1]])
       
    plt.rcParams['figure.figsize'] = 8, 8
    fig, ax=plt.subplots()
    ax = plt.gca()
    ax.set_facecolor('xkcd:dark sand') #  warm purple
    ax.axes.get_xaxis().set_visible(False)
    ax.axes.get_yaxis().set_visible(False)
    mn=min(min(zlep_x),min(zlep_y))
    mx=max(max(zlep_x),max(zlep_y))
    dm=np.rint((mx-mn)/10)
    h = ax.axis([mn-dm,mx+dm,mn-dm,mx+dm])
    masa=len(zlep_x)
    licz_kol=10  # liczba kolorow
    mfazy=int(masa/licz_kol)

    tabkol=np.fromfunction(lambda i:np.floor(i/(mfazy+1)) ,(masa,),dtype=int)

    zb2=ax.scatter(zlep_x,zlep_y,s=1,c=tabkol,cmap=cm.nipy_spectral)
    #cm.hsv, cm.nipy_spectral, cm.rainbow, , cm.gist_rainbow
    tyt="Zlepek Edena o masie ="+str(len(zlep_x))
    ax.set_title(tyt)
    plt.axis('equal')
    plt.show()
