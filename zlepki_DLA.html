Dyfuzyjny wzrost zlepków¶

Symulacje dyfuzyjnego wzrostu zlepka możemy przeprowadzić na sieci kwadratowej w następujący sposób

\begin{itemize} \item W środku sieci umieszczamy zarodek zlepka. \item Cztery sąsiednie węzły wokół zarodka tworzą początkowy obwód zlepka. \item Do zlepka będą dołączane cząstki, które poruszając się losowo po węzłach sieci dotrą do obwodu zlepka. \item Miejscem startu każdej z cząstek może być losowe położenie na okręgu o promieniu R, którego środek leży w środku sieci. \item Węzły brzegowe sieci można potraktować jako węzły pochłaniające, co powoduje, że losowy spacer każdej z cząstek może zakończyć się albo na brzegu sieci albo dołączeniem do zlepka. \item Symulację kończymy gdy zlepek osiągnie zadaną z góry masę (liczbę cząstek) lub gdy jego rozmiar liniowy będzie bliski 2R. \end{itemize}

import random
import numpy as np
import matplotlib.pyplot as plt

v=np.array([[1, 0],[0, 1],[-1, 0],[0,-1]]) # 4 wektory do najbliższych sąsiadów

Lp=300  # współrzędna środka sieci 
L=2*Lp  # rozmiar liniowy sieci
siec=np.zeros((L,L),int)  # siec o  L*L pocżatkowo wyzerowana
#  Składowe sieci mogą przyjmować wartości -1, 0,1 i 2
# Wartości te reprezentują
#  -1 węzły obwodu
#   0 węzły dyfuzyjne
#   1 węzły zlepka
#   2 węzły pochlaniające

def init():
    siec[Lp,Lp]=1  #zarodek
    mod_obwodu(Lp,Lp)
    #wezly pochlaniajace
    for i in range(L):
        siec[i,0]=2
        siec[i,L-1]=2
        siec[0,i]=2
        siec[L-1,i]=2
    return np.array([Lp,Lp],int)


def start_cz(R):
    kat=2*np.pi*random.random()
    # losowe położenie początkowe na okręgu o promieniu R
    return np.array([Lp+np.rint(R*np.cos(kat)),Lp+np.rint(R*np.sin(kat))],int)
       
def mod_obwodu(x,y):
    for i in range(4):
        if siec[x+v[i,0],y+v[i,1]]==0:    
            siec[x+v[i,0],y+v[i,1]]=-1

def spacer_los(R): 
    #położenie początkowe na okręgu o promieniu R
    r=start_cz(R)
    while siec[r[0],r[1]]==0:
        r=r+v[random.randint(0,3)] #losowy  krok
    return siec[r[0],r[1]], r 

def wzrost(R,mmax,xy_z):
    licz=1
    rb=init()
    xy_z.append(rb)
    while licz<mmax:        
        status,rb=spacer_los(R)
        if status==-1:
            siec[rb[0],rb[1]]=1
            mod_obwodu(rb[0],rb[1])
            xy_z.append(rb)
            licz+=1
        
    

def rys_zlep(kn,xy):
    fig, ax=plt.subplots()
    xt, yt = zip(*xy)
    xf=xt[:kn]
    yf=yt[:kn]
    mn=min(min(xf),min(yf))
    mx=max(max(xf),max(yf))
    dm=np.rint((mx-mn)/4)
    if dm <1:
        dm=1
    ax.set_xticks(np.arange(mn-dm,mx+dm,dm))
    ax.set_yticks(np.arange(mn-dm,mx+dm,dm))
    zlep=ax.scatter(xf,yf,s=6,c='r')
    ax.set_ylabel('y')
    ax.set_xlabel('x')
    tytul = "Zlepek o masie =" +str(kn)
    ax.set_title(tytul)
    plt.axis('equal')
    plt.show()

Wzrost zlepka na obwodzie koła.¶

Zarodki rozmieszczone są na okręgu o promieniu R. Kolejne cząstki startują ze środka koła, poruszając się losowo po węzłach sieci. Gdy cząstka dotrze do obwodu zlepka zostaje do niego przyłączona.

#   1 węzły zlepka


def initC(Rc):
    Rc2=Rc**2
    for i in range(LC):
        for k in range(LC):
            if ((i-LCp)**2+(k-LCp)**2)>=Rc2:
                circle[i,k]=1  #zarodek
    for i in range(1,LC-1):
        for k in range(1,LC-1):
            if n_sas(i,k)>0:
                circle[i,k]=-1
    
def n_sas(x,y):
    licz=0
    if circle[x,y]==0:
        for i in range(4):
            if circle[x+v[i,0],y+v[i,1]]==1:
                licz+=1
    return licz
        

def start_C():
    return np.array([LCp,LCp],int)
       
def mod_obwC(x,y):

Model Edena¶

Prosty model zaproponowany przez Edena do symulacji wzrostu kolonii komórkowych.

W tym modelu zaczynamy od umieszczenia zarodka w środku sieci. Niezajęte sąsiednie węzły tworzą obwód zlepka. W najprostszej wersji modelu nowa cząstka jest dołączana do losowo wybranego węzła z obwodu zlepka. Dołączenie nowej cząstki modyfikuje obwód. Taki proces wzrostu jest powtarzany tak długo aż zlepek osiągnie brzeg sieci.

def rys_zlep(kn):
#4 wektory do najbliższych sąsiadów
    v=np.array([[1, 0],[0, 1],[-1, 0],[0,-1]])
    Lp=500 
    L=2*Lp
    siec=np.zeros((L,L),int)

    zlep_x=[]
    zlep_y=[]
#zarodek
    r=np.array([0,0])
    lista=[[Lp,Lp]]
    for k in range (kn):
        r=random.choice(lista)
        if (r[0]==0)or(r[0]==L-1):
            break
        if (r[1]==0)or(r[1]==L-1):
            break    
        siec[r[0],r[1]]=1
        zlep_x.append(r[0])
        zlep_y.append(r[1])
        for i in range(4):
            rs=r+v[i]
            if(siec[rs[0],rs[1]]==0):
                lista.append([rs[0],rs[1]])
        lista.remove([r[0],r[1]])
       
    plt.rcParams['figure.figsize'] = 8, 8
    fig, ax=plt.subplots()
    ax = plt.gca()
    ax.set_facecolor('xkcd:dark sand') #  warm purple
    ax.axes.get_xaxis().set_visible(False)
    ax.axes.get_yaxis().set_visible(False)
    mn=min(min(zlep_x),min(zlep_y))
    mx=max(max(zlep_x),max(zlep_y))
    dm=np.rint((mx-mn)/10)
    h = ax.axis([mn-dm,mx+dm,mn-dm,mx+dm])
    masa=len(zlep_x)
    licz_kol=10  # liczba kolorow
    mfazy=int(masa/licz_kol)

    tabkol=np.fromfunction(lambda i:np.floor(i/(mfazy+1)) ,(masa,),dtype=int)

    zb2=ax.scatter(zlep_x,zlep_y,s=1,c=tabkol,cmap=cm.nipy_spectral)
    #cm.hsv, cm.nipy_spectral, cm.rainbow, , cm.gist_rainbow
    tyt="Zlepek Edena o masie ="+str(len(zlep_x))
    ax.set_title(tyt)
    plt.axis('equal')
    plt.show()
